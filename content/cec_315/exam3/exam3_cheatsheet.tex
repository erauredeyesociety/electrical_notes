\section*{Transforms \& ROC}

\textbf{Laplace.} $X(s)=\!\int x(t)e^{-st}dt$, $s=\sigma+j\omega$. Unilateral: lower limit $0^-$. \textbf{FT exists iff $j\omega$-axis $\subset$ ROC.}

\textbf{Z.} $X(z)=\!\sum x[n]z^{-n}$, $z=re^{j\Omega}$. Unilateral: $n\ge 0$. \textbf{DTFT exists iff unit circle $\subset$ ROC.}

\textbf{ROC rules (both).} Never contains poles.
\begin{itemize}
\item \textbf{Right-sided:} $\sigma>\sigma_{\max}$ / $|z|>r_{\max}$.
\item \textbf{Left-sided:} $\sigma<\sigma_{\min}$ / $|z|<r_{\min}$.
\item \textbf{Two-sided:} vertical strip / annulus.
\item \textbf{Finite duration:} entire $s$- / $z$-plane (exc.\ $0,\infty$).
\item \textbf{Causal + stable} $\Leftrightarrow$ LHP poles (CT) / $|p|<1$ (DT).
\end{itemize}

\section*{Partial Fraction Expansion}

\textbf{Distinct.} $X=\sum A_k/(s-p_k)$, cover-up: $A_k=(s-p_k)X|_{s=p_k}$.

\textbf{Repeated} (order $r$ at $p$): $\sum_{k=1}^r B_k/(s-p)^k$ with
\[
B_k=\tfrac{1}{(r-k)!}\tfrac{d^{r-k}}{ds^{r-k}}\!\bigl[(s-p)^r X\bigr]_{s=p}.
\]

\textbf{Complex conjugate} $p=-a\pm j\omega_d$: complete the square $(s+a)^2+\omega_d^2$, then use $\cos/\sin$ pairs.

\textbf{Z:} always in $z^{-1}$ (NOT $z$); ROC selects right- vs.\ left-sided for each pole.

\textbf{Improper rationals (both).} If $\deg(\text{num})\ge\deg(\text{den})$: long-divide first.
\begin{itemize}
\item \textbf{Laplace:} constant remainder $\to \delta(t)$; higher polynomial $\to \delta^{(k)}(t)$; PFE the strictly-proper part.
\item \textbf{Z:} polynomial in $z^{-1}$ $\to$ finite-duration samples (scaled $\delta[n-k]$); PFE the strictly-proper part in $z^{-1}$.
\end{itemize}

\section*{System Analysis}

\textbf{CT (ODE).} $y^{(k)}\!\to\! s^k Y-\sum_{j=0}^{k-1}\! s^{k-1-j} y^{(j)}(0^-)$. Solve $Y=Y_\text{ZS}+Y_\text{ZI}$; PFE; invert. $H(s)=Y_\text{ZS}/X$.

\textbf{DT (diff.\ eq.).} $y[n-k]\to z^{-k}Y+\sum_{m=1}^k z^{-(k-m)}y[-m]$. Solve; PFE in $z^{-1}$; invert. $H(z)=Y_\text{ZS}/X$.

Total response = \textbf{ZSR} (ICs $=0$, input only) $+$ \textbf{ZIR} (input $=0$, ICs only).

\section*{Sampling (Lec 22)}

\[
X_p(j\omega)=\tfrac{1}{T}\!\sum_k X\!\bigl(j(\omega-k\omega_s)\bigr),\quad \omega_s=\tfrac{2\pi}{T}
\]

\textbf{Theorem.} Bandlimited to $\omega_M \Rightarrow$ recover iff $\omega_s > 2\omega_M$ (\emph{strict}). Nyquist rate $= 2\omega_M$. $\omega_s=2\omega_M$ can fail (sine at zero crossings).

\textbf{Reconstruct.} LPF gain $T$, cutoff $\omega_c \in (\omega_M, \omega_s-\omega_M)$.
\[
x_r(t)=\sum_n x(nT)\,\sinc\!\Bigl(\tfrac{t-nT}{T}\Bigr)
\]
\textbf{ZOH:} $H_0(j\omega) = \tfrac{2\sin(\omega T/2)}{\omega}e^{-j\omega T/2}$ (sinc droop, imperfect).

\textbf{Aliasing} (irreversible, use AAF \emph{before} sampler):
\begin{itemize}
\item Shortcut ($f_s/2 < f_0 < f_s$): $f_\text{alias} = f_s - f_0$.
\item Rad: pick $k$ so $|\omega_0 - k\omega_s| \le \omega_s/2$.
\item Sign flip $\Rightarrow$ reversed apparent rotation (helicopter blade).
\end{itemize}

\textbf{Traps.} $\omega_s = 2\omega_M$ not enough. $x(t-t_0)$ keeps Nyquist rate; $x(at)$ scales by $|a|$. C/D map: $\Omega = \omega T$.

\textbf{Bandwidth of products.} $x_1(t)x_2(t)$ has BW $\le \omega_{M,1}+\omega_{M,2}$. Special case: squaring doubles BW. Nyquist rate of product $\le 2(\omega_{M,1}+\omega_{M,2})$.

\section*{Feedback (Lec 23)}

\[
T(s) = \frac{G}{1+GH}\ \text{(neg)}; \quad \frac{G}{1-GH}\ \text{(pos)}
\]

Unity fdbk: $T = G/(1+G)$. Loop gain $L = GH$. \textbf{Closed-loop poles} $=$ roots of $1+L = 0$ (characteristic equation).

\textbf{Nyquist criterion.} $N = Z - P$ where $N$ is CW encirclements of $-1$, $P$ is OL RHP poles of $L$, $Z$ is CL RHP poles; need $Z=0$. If OL stable ($P=0$): CL stable iff Nyquist plot of $L(j\omega)$ does \emph{not} encircle $-1$ (or $-1/K$).

\textbf{Gain margin} at $\omega_{180}$ (where $\angle L = -180^\circ$):
\[
\mathrm{GM}_\mathrm{dB} = -20\log_{10}|L(j\omega_{180})|
\]

\textbf{Phase margin} at $\omega_\text{gc}$ (where $|L| = 1$):
\[
\mathrm{PM} = 180^\circ + \angle L(j\omega_\text{gc})
\]

Both margins $> 0 \Rightarrow$ stable. Max added delay: $\tau_{\max} = \mathrm{PM}_\mathrm{rad}/\omega_\text{gc}$. First-order plants: $\mathrm{GM} = \infty$ (phase never hits $-180^\circ$).

\textbf{Routh--Hurwitz} for $a_n s^n + \cdots + a_0$: number of sign changes in the first column $=$ number of RHP roots. Necessary (not sufficient): all $a_i > 0$, none missing. 2nd-order $s^2 + bs + c$: stable iff $b, c > 0$.

\section*{Procedures (Step-by-Step)}

\textbf{Inverse Laplace.} Proper? $\to$ PFE in $s$ (distinct / repeated / complex-conj; complete square) $\to$ ROC picks direction per pole $\to$ sum table entries.

\textbf{Inverse Z.} Improper? $\to$ long-divide $\to$ PFE in $z^{-1}$ $\to$ ROC outside $|d|\to d^n u[n]$, inside $\to -d^n u[-n-1]$ $\to$ complex-conj $p=re^{\pm j\Omega_0}$: use $r^n \cos/\sin$ pair.

\textbf{Unilateral L IVP.} Transform ICs (\textbf{minus}) $\to$ solve $Y$ $\to$ split ZSR $+$ ZIR $\to$ PFE $\to$ invert $\to$ verify $y(0^+)$, $y(\infty)$.

\textbf{Unilateral Z diff eqn.} Transform ICs (\textbf{plus}) $\to$ solve $Y$ $\to$ PFE in $z^{-1}$ $\to$ invert $\to$ check $y[0], y[1]$ vs.\ difference equation directly.

\textbf{Block diagram.} Innermost loop $\to H/(1+GH)$ $\to$ replace with single block $\to$ combine cascades (multiply) and parallel (add) $\to$ poles $=$ roots of $1+GH=0$.

\textbf{Aliased frequency.} Reduce $f_0 \bmod f_s$; fold to $[0, f_s/2]$. Shortcut for $f_s/2 < f_0 < f_s$: $f_\text{alias} = f_s - f_0$.

\textbf{GM/PM from Bode.} Read $\omega_\text{gc}$ at 0 dB $\to$ PM from phase. Read $\omega_{pc}$ at $-180^\circ$ $\to$ GM from $|L|$. $\tau_{\max} = \mathrm{PM}_\mathrm{rad}/\omega_\text{gc}$.

\section*{Top Traps}

\begin{itemize}
\item \textbf{Always state ROC.} Same $X$, different ROC $=$ different signal.
\item Unilateral L: \textbf{minus} $y(0^-)$. Unilateral Z: \textbf{plus} $y[-1]$. Opposite signs.
\item Repeated pole order $n$ needs $n$ PFE terms.
\item DT stability: $|a| < 1$, not $a < 1$. Pole at $z = -0.9$ IS stable.
\item FVT invalid if system unstable or marginal.
\item $\omega_s > 2\omega_M$ strict; aliasing irreversible; AAF \emph{before} sampler.
\item GM from $|L|$ at $-180^\circ$; PM from $\angle L$ at 0 dB. Don't swap.
\item Poles of $\tfrac{1}{1-az^{-1}}$ at $z = a$, not $1/a$.
\item PFE in Z: use $z^{-1}$, never $z$.
\item RHP \emph{zero} $\ne$ instability (only poles matter).
\item Marginal poles $\Rightarrow$ \emph{not} BIBO stable.
\item Negative fdbk $\to 1+GH$; positive $\to 1-GH$.
\item Negative feedback does \emph{not} always stabilize (too much gain $\to$ unstable).
\end{itemize}
