\section*{Transform Definitions}
\footnotesize
\begin{tabular}{@{}l l@{}}
\toprule
\textbf{Bilat.\ Laplace} & $X(s)=\int_{-\infty}^{\infty} x(t)e^{-st}\,dt$ \\
Inverse & $x(t)=\tfrac{1}{2\pi j}\int_{\sigma-j\infty}^{\sigma+j\infty}\! X(s)e^{st}\,ds$ \\
\textbf{Unilat.\ Laplace} & $\mathcal{X}(s)=\int_{0^-}^{\infty} x(t)e^{-st}\,dt$ \\
\textbf{Bilat.\ Z} & $X(z)=\sum_{n=-\infty}^{\infty} x[n]z^{-n}$ \\
Inverse & $x[n]=\tfrac{1}{2\pi j}\oint X(z)z^{n-1}\,dz$ \\
\textbf{Unilat.\ Z} & $\mathcal{X}(z)=\sum_{n=0}^{\infty} x[n]z^{-n}$ \\
\bottomrule
\end{tabular}

$s=\sigma+j\omega$, $z=re^{j\Omega}$. FT exists iff $j\omega$-axis $\subset$ ROC; DTFT exists iff unit circle $\subset$ ROC.

\section*{Laplace ROC Rules}
\footnotesize
\begin{itemize}
\item Vertical strip; never contains poles.
\item Finite duration $\Rightarrow$ entire $s$-plane.
\item Right-sided (causal): $\Re\{s\}>\sigma_{\max}$.
\item Left-sided (anticausal): $\Re\{s\}<\sigma_{\min}$.
\item Two-sided: strip between poles (may be empty).
\item Rational + causal $\Leftrightarrow M\le N$, right-sided.
\item \textbf{BIBO stable} $\Leftrightarrow j\omega$-axis $\in$ ROC.
\item \textbf{Causal+Stable} $\Leftrightarrow$ all poles $\Re\{p_i\}<0$.
\end{itemize}

\section*{Z ROC Rules}
\footnotesize
\begin{itemize}
\item Annulus $r_1<|z|<r_2$; never contains poles.
\item Finite duration $\Rightarrow$ entire $z$-plane (maybe excl.\ $0,\infty$).
\item Right-sided (causal): $|z|>r_{\max}$ (incl.\ $z=\infty$ if causal).
\item Left-sided (anticausal): $|z|<r_{\min}$.
\item Two-sided: ring between poles.
\item \textbf{BIBO stable} $\Leftrightarrow$ unit circle $\subset$ ROC.
\item \textbf{Causal+Stable} $\Leftrightarrow$ all poles $|p_i|<1$.
\item LHP (CT) $\leftrightarrow$ interior of unit circle (DT).
\end{itemize}

\section*{Partial Fraction Expansion}
\footnotesize
\textbf{Distinct:} $X(s)=\sum_k\tfrac{A_k}{s-p_k}$, \ $A_k=(s-p_k)X(s)\big|_{s=p_k}$.

\textbf{Repeated} (order $r$ at $p$): $\sum_{k=1}^{r}\tfrac{B_k}{(s-p)^k}$,
\[
B_k=\tfrac{1}{(r-k)!}\tfrac{d^{r-k}}{ds^{r-k}}\!\left[(s-p)^r X(s)\right]_{s=p}.
\]

\textbf{Complex conj.\ pair} $p=-a\pm j\omega_d$: complete the square; use $(s+a)^2+\omega_d^2$ with $\cos/\sin$ pairs.

\textbf{Z-transform:} expand in $z^{-1}$ (NOT $z$). Improper $X(z)$: long-divide first. ROC selects right- vs.\ left-sided for each term.

\section*{Unilateral System Analysis}
\footnotesize
\textbf{CT (ODE):} $y^{(k)}\to s^k Y-\sum_{j=0}^{k-1} s^{k-1-j} y^{(j)}(0^-)$. Solve $Y(s)=Y_{\text{zs}}+Y_{\text{zi}}$; PFE; invert. $H(s)=Y_{\text{zs}}/X$.

\textbf{DT (diff.\ eq.):} $y[n-k]\to z^{-k}Y+\sum_{m=1}^{k} z^{-(k-m)}y[-m]$. Solve, PFE in $z^{-1}$, invert. $H(z)=Y_{\text{zs}}/X$.

Total $=$ ZSR (ICs$=0$) $+$ ZIR (input$=0$).

\section*{Sampling (Lec 22)}
\footnotesize
\begin{itemize}
\item Impulse train $p(t)=\sum_n\delta(t-nT)$; $\omega_s=2\pi/T$, $f_s=1/T$.
\item $X_p(j\omega)=\tfrac{1}{T}\sum_{k} X(j(\omega-k\omega_s))$.
\item \textbf{Sampling Thm:} bandlimited to $\omega_M \Rightarrow$ unique recovery iff $\omega_s>2\omega_M$ (strict). Nyquist rate $=2\omega_M$.
\item Aliasing: $f_{\text{alias}}=|f_s-f_0|$ for $f_0\in(f_s/2,f_s)$; irreversible---use anti-alias LPF first.
\item \textbf{Ideal reconstruction:} LPF gain $T$, cutoff $\omega_c\in(\omega_M,\omega_s-\omega_M)$ (typ.\ $\omega_s/2$).
\item \textbf{Sinc interp.:} $x_r(t)=\sum_n x(nT)\,\sinc\!\left(\tfrac{t-nT}{T}\right)$, $\sinc u=\tfrac{\sin\pi u}{\pi u}$.
\item \textbf{ZOH:} conv.\ with rect of width $T$; $H_0(j\omega)=\tfrac{2\sin(\omega T/2)}{\omega}e^{-j\omega T/2}$.
\item \textbf{C/D map:} $\Omega=\omega T$.
\item Traps: $\omega_s=2\omega_M$ not enough; squaring doubles BW; time-shift preserves Nyquist rate, $x(at)$ scales it by $|a|$.
\end{itemize}

\section*{Feedback Systems (Lec 23)}
\footnotesize
\textbf{Negative feedback} (forward $G$, fdbk $H$):
\[
T(s)=\frac{G(s)}{1+G(s)H(s)},\qquad \text{unity: } T=\frac{G}{1+G}.
\]
Positive fdbk: $G/(1-GH)$. Loop gain $L=GH$. Closed-loop poles $=$ roots of $1+L(s)=0$ (characteristic eq.).

\textbf{Nyquist criterion:} $N=Z-P$, with $N=$ CW encirclements of $-1$, $P=$ open-loop RHP poles of $L$, $Z=$ closed-loop RHP poles; need $Z=0$. If plant open-loop stable ($P=0$): closed-loop stable iff Nyquist plot of $L(j\omega)$ does not encircle $-1$ (or $-1/K$).

\textbf{Gain margin} at $\omega_{180}$ (where $\angle L=-180^\circ$):
\[
\mathrm{GM}_{\mathrm{dB}}=-20\log_{10}|L(j\omega_{180})|.
\]
\textbf{Phase margin} at $\omega_{\text{gc}}$ (where $|L|=1$):
\[
\mathrm{PM}=180^\circ+\angle L(j\omega_{\text{gc}}).
\]
Both $>0 \Rightarrow$ stable. Delay tolerance $\tau_{\max}=\mathrm{PM}_{\text{rad}}/\omega_{\text{gc}}$. First-order plants: $\mathrm{GM}=\infty$ (phase never hits $-180^\circ$).

\textbf{Routh--Hurwitz} for $a_n s^n+\cdots+a_0$: build Routh array; \# sign changes in first column $=$ \# RHP roots. Necessary (not sufficient): all $a_i>0$, none missing.

\section*{One-Liners / Traps}
\footnotesize
\begin{itemize}
\item Unil.\ Laplace diff: $sX-x(0^-)$ (\textbf{minus}). Unil.\ Z shift: $z^{-1}X+x[-1]$ (\textbf{plus}).
\item Conv $\leftrightarrow$ multiply. ROC product $\supseteq R_x\cap R_y$.
\item Causal+Stable $\Leftrightarrow$ LHP poles (CT) / $|p|<1$ (DT).
\item Poles of $1/(1-az^{-1})$ at $z=a$, not $1/a$.
\item PFE in Z: use $z^{-1}$, not $z$.
\item Always state ROC; same $X(s)$, different ROC $=$ different signals.
\item FVT invalid if growing/oscillating.
\item RHP \emph{zero} $\ne$ instability (only poles matter).
\item Marginal poles ($j\omega$-axis / unit circle) $\Rightarrow$ \emph{not} BIBO stable.
\item $f_{\text{alias}}=|f_s-f_0|$; $\omega_s>2\omega_M$ strict.
\item $T=G/(1+GH)$; char.\ eq.\ $1+GH=0$.
\item Negative fdbk $\to 1+GH$; positive fdbk $\to 1-GH$.
\end{itemize}
