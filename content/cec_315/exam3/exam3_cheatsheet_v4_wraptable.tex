\section*{Transforms \& ROC + Laplace Pairs}

\begin{wraptable}[15]{l}{0pt}
  \begin{tabular}{@{}l l l@{}}
    \toprule
    $x(t)$ & $X(s)$ & ROC \\
    \midrule
    $\delta(t)$ & $1$ & all $s$ \\
    $\delta(t{-}t_0)$ & $e^{-st_0}$ & all $s$ \\
    $u(t)$ & $1/s$ & $\Re\{s\}{>}0$ \\
    $tu(t)$ & $1/s^2$ & $\Re\{s\}{>}0$ \\
    $\tfrac{t^{n-1}}{(n-1)!}u(t)$ & $1/s^n$ & $\Re\{s\}{>}0$ \\
    $e^{-at}u(t)$ & $\tfrac{1}{s{+}a}$ & $\Re\{s\}{>}{-}a$ \\
    $te^{-at}u(t)$ & $\tfrac{1}{(s{+}a)^2}$ & $\Re\{s\}{>}{-}a$ \\
    $\tfrac{t^{n-1}}{(n-1)!}e^{-at}u$ & $\tfrac{1}{(s{+}a)^n}$ & $\Re\{s\}{>}{-}a$ \\
    $\cos(\omega_0 t)u$ & $\tfrac{s}{s^2{+}\omega_0^2}$ & $\Re\{s\}{>}0$ \\
    $\sin(\omega_0 t)u$ & $\tfrac{\omega_0}{s^2{+}\omega_0^2}$ & $\Re\{s\}{>}0$ \\
    $e^{-at}\!\cos(\omega_d t)u$ & $\tfrac{s{+}a}{(s{+}a)^2{+}\omega_d^2}$ & $\Re\{s\}{>}{-}a$ \\
    $e^{-at}\!\sin(\omega_d t)u$ & $\tfrac{\omega_d}{(s{+}a)^2{+}\omega_d^2}$ & $\Re\{s\}{>}{-}a$ \\
    $-e^{-at}u(-t)$ & $\tfrac{1}{s{+}a}$ & $\Re\{s\}{<}{-}a$ \\
    \bottomrule
  \end{tabular}
\end{wraptable}
\textbf{Laplace.} $X(s)=\!\int x(t)e^{-st}dt$, $s=\sigma+j\omega$. Unilateral: lower limit $0^-$. FT exists iff $j\omega$-axis $\subset$ ROC.
\textbf{Z.} $X(z)=\!\sum x[n]z^{-n}$, $z=re^{j\Omega}$. Unilateral: $n\ge 0$. DTFT exists iff unit circle $\subset$ ROC.
\textbf{ROC rules (both).} Never contain poles. Right-sided: $\sigma>\sigma_{\max}$ / $|z|>r_{\max}$. Left: reverse. Two-sided: strip / annulus. Finite duration: all $s$/$z$ except $0,\infty$. \textbf{Causal+stable} $\Leftrightarrow$ LHP poles (CT) / $|p|<1$ (DT).
\textbf{PFE --- Distinct:} $\sum A_k/(s-p_k)$ with $A_k=(s-p_k)X|_{p_k}$.
\textbf{Repeated} order $r$ at $p$: $\sum B_k/(s-p)^k$ with $B_k=\tfrac{1}{(r-k)!}\tfrac{d^{r-k}}{ds^{r-k}}[(s-p)^r X]|_p$.
\textbf{Complex conj.} $p=-a\pm j\omega_d$: complete the square $(s+a)^2+\omega_d^2$, use $\cos/\sin$ pairs.
\textbf{Z:} always in $z^{-1}$ (never $z$); ROC selects right/left per pole.
\textbf{Improper rationals.} Long-divide first. Laplace: const remainder $\to\delta(t)$; polynomial $\to\delta^{(k)}(t)$. Z: polynomial in $z^{-1}$ $\to$ finite-duration samples $\delta[n{-}k]$. PFE the strictly-proper part either way.

\section*{Z Pairs + Properties}

\begin{wraptable}[10]{l}{0pt}
  \begin{tabular}{@{}l l l@{}}
    \toprule
    $x[n]$ & $X(z)$ & ROC \\
    \midrule
    $\delta[n]$ & $1$ & all $z$ \\
    $\delta[n{-}k]$ & $z^{-k}$ & $z\ne 0$ \\
    $u[n]$ & $\tfrac{1}{1-z^{-1}}$ & $|z|{>}1$ \\
    $-u[-n{-}1]$ & $\tfrac{1}{1-z^{-1}}$ & $|z|{<}1$ \\
    $a^n u[n]$ & $\tfrac{1}{1-az^{-1}}$ & $|z|{>}|a|$ \\
    $-a^n u[-n{-}1]$ & $\tfrac{1}{1-az^{-1}}$ & $|z|{<}|a|$ \\
    $na^n u[n]$ & $\tfrac{az^{-1}}{(1-az^{-1})^2}$ & $|z|{>}|a|$ \\
    $(n{+}1)a^n u[n]$ & $\tfrac{1}{(1-az^{-1})^2}$ & $|z|{>}|a|$ \\
    \bottomrule
  \end{tabular}
\end{wraptable}
\textbf{Linearity}: $aX_1+bX_2$. \textbf{Time shift}: $x[n{-}n_0]\to z^{-n_0}X$. \textbf{$z$-scaling}: $z_0^n x[n]\to X(z/z_0)$. \textbf{Reversal}: $x[-n]\to X(z^{-1})$. \textbf{Diff in $z$}: $nx[n]\to -z\,dX/dz$. \textbf{First diff}: $(1-z^{-1})X$. \textbf{Accumulation}: $X/(1-z^{-1})$. \textbf{Convolution}: $x_1*x_2\to X_1X_2$.
\textbf{Unilateral shift-1:} $x[n{-}1]\to z^{-1}X+x[-1]$ \textbf{(plus)}. \textbf{Shift-2:} $z^{-2}X+x[-1]z^{-1}+x[-2]$.
\textbf{IVT}: $x[0]=\lim_{z\to\infty}X$. \textbf{FVT}: $x[\infty]=\lim_{z\to 1}(1-z^{-1})X$ (valid iff $(1-z^{-1})X$ poles inside unit circle).
\textbf{Damped DT sinusoids} ($|z|>r$): $r^n\cos(\Omega_0 n)u \leftrightarrow \tfrac{1-r\cos\Omega_0 z^{-1}}{1-2r\cos\Omega_0 z^{-1}+r^2 z^{-2}}$; $r^n\sin(\Omega_0 n)u \leftrightarrow \tfrac{r\sin\Omega_0 z^{-1}}{1-2r\cos\Omega_0 z^{-1}+r^2 z^{-2}}$. Set $r=1$ for undamped.

\section*{Laplace Properties + System Analysis}

\begin{wraptable}[14]{l}{0pt}
  \begin{tabular}{@{}l l@{}}
    \toprule
    Linearity & $aX{+}bY$ \\
    Time shift & $x(t{-}t_0)\!\to\!e^{-st_0}X$ \\
    $s$-shift & $e^{s_0 t}x\!\to\!X(s{-}s_0)$ \\
    Scaling & $x(at)\!\to\!\tfrac{1}{|a|}X(s/a)$ \\
    Diff.\ bilat. & $x'\!\to\!sX$ \\
    \textbf{Unil.\ diff} & $sX{-}x(0^-)$ \textbf{(--)} \\
    $x''$ unilat. & $s^2X{-}sx(0^-){-}x'(0^-)$ \\
    Integ. & $\int_{-\infty}^t\!x\,d\tau\!\to\!X/s$ \\
    Diff.\ in $s$ & $-tx\!\to\!dX/ds$ \\
    Conv. & $x*y\!\to\!XY$ \\
    IVT & $\lim_{s\to\infty}\!sX$ \\
    FVT$^\ast$ & $\lim_{s\to 0}\!sX$ \\
    \bottomrule
  \end{tabular}\\[1pt]
  {\footnotesize$^\ast$ poles of $sX$ in LHP}
\end{wraptable}
\textbf{CT (ODE).} $y^{(k)}\!\to\! s^k Y - \sum_{j=0}^{k-1}\! s^{k-1-j} y^{(j)}(0^-)$. Solve $Y=Y_{\text{ZS}}+Y_{\text{ZI}}$; PFE; invert. $H(s)=Y_{\text{ZS}}/X$.
\textbf{DT (diff.\ eq.).} $y[n{-}k]\to z^{-k}Y + \sum_{m=1}^k z^{-(k-m)}y[-m]$. Solve; PFE in $z^{-1}$; invert. $H(z)=Y_{\text{ZS}}/X$.
\textbf{Total response} = ZSR (ICs$=0$, input only) $+$ ZIR (input$=0$, ICs only).
\textbf{Unilateral Laplace IVP.} Transform ICs (\textbf{minus}) $\to$ solve $Y$ $\to$ split ZSR+ZIR $\to$ PFE $\to$ invert $\to$ verify $y(0^+)$, $y(\infty)$.
\textbf{Unilateral Z diff eqn.} Transform ICs (\textbf{plus}) $\to$ solve $Y$ $\to$ PFE in $z^{-1}$ $\to$ invert $\to$ check $y[0],y[1]$ from diff eqn directly.
\textbf{Sign trap.} Unil.\ L: $\mathbf{-}y(0^-)$. Unil.\ Z: $\mathbf{+}y[-1]$. Opposite signs.

\section*{Sampling (Lec 22)}

\begin{wraptable}[10]{r}{0pt}
  \begin{tabular}{@{}l l@{}}
    \toprule
    $X_p$ & $\tfrac{1}{T}\!\sum_k X(j(\omega{-}k\omega_s))$ \\
    $\omega_s$ & $2\pi/T$, $f_s{=}1/T$ \\
    Nyquist rate & $2\omega_M$ \\
    Theorem & $\omega_s{>}2\omega_M$ (strict) \\
    Recon LPF & gain $T$, cut $(\omega_M,\omega_s{-}\omega_M)$ \\
    Sinc recon & $\sum x(nT)\sinc(\tfrac{t{-}nT}{T})$ \\
    ZOH & $\tfrac{2\sin(\omega T/2)}{\omega}e^{-j\omega T/2}$ \\
    Alias & $|f_s{-}f_0|$, $f_s/2{<}f_0{<}f_s$ \\
    C/D map & $\Omega{=}\omega T$ \\
    \bottomrule
  \end{tabular}
\end{wraptable}
\textbf{Theorem.} Bandlimited to $\omega_M$ $\Rightarrow$ perfect recovery iff $\omega_s>2\omega_M$ (\emph{strict}). $\omega_s=2\omega_M$ can fail (sine at zero crossings).
\textbf{Reconstruction.} Pass $X_p$ through LPF with passband gain $T$ and cutoff in $(\omega_M,\omega_s{-}\omega_M)$; equivalent time-domain: sinc interpolation. ZOH is NOT exact --- introduces sinc-shaped droop.
\textbf{Aliasing is irreversible.} Use anti-aliasing LPF \emph{before} the sampler, never after. For a pure tone $f_0>f_s/2$: aliased frequency is $|f_s-f_0|$; sign flip means reversed apparent direction (helicopter blade).
\textbf{Nyquist rate transforms.} Time-shift $x(t-t_0)$ keeps rate. Time-scale $x(at)$ scales rate by $|a|$.
\textbf{Bandwidth of products.} $x_1 x_2$ has BW $\le \omega_{M,1}+\omega_{M,2}$; squaring doubles BW. Nyquist rate of product $\le 2(\omega_{M,1}+\omega_{M,2})$.

\section*{Feedback (Lec 23)}

\begin{wraptable}[11]{l}{0pt}
  \begin{tabular}{@{}l l@{}}
    \toprule
    Neg fdbk & $T{=}\tfrac{G}{1+GH}$ \\
    Pos fdbk & $T{=}\tfrac{G}{1-GH}$ \\
    Unity fdbk & $T{=}G/(1{+}G)$ \\
    Loop gain & $L{=}GH$ \\
    CL poles & roots of $1{+}L{=}0$ \\
    Unstable & $L{=}{-}1$ \\
    PM & $180^\circ{+}\angle L(\omega_{gc})$ \\
    GM$_\mathrm{dB}$ & ${-}|L(\omega_{pc})|_\mathrm{dB}$ \\
    Max delay & $\mathrm{PM}_\mathrm{rad}/\omega_{gc}$ \\
    2nd-ord Routh & $b,c{>}0$ \\
    \bottomrule
  \end{tabular}
\end{wraptable}
Closed-loop poles = roots of $1+GH=0$. Cascade multiplies, parallel adds, nested loops reduce innermost first.
\textbf{Nyquist criterion.} $N=Z-P$: $N$ CW encirclements of $-1$, $P$ open-loop RHP poles of $L$, $Z$ closed-loop RHP poles. Need $Z=0$. If open-loop stable ($P=0$): closed-loop stable iff Nyquist plot does not encircle $-1$ (or $-1/K$).
\textbf{Gain/phase margins.} $\omega_{gc}$ is where $|L|=1$; PM is phase lead there. $\omega_{pc}$ is where $\angle L=-180^\circ$; GM is magnitude-below-0-dB there. Both $>0 \Rightarrow$ stable. Delay tolerance $\tau_{\max}=\mathrm{PM}_\mathrm{rad}/\omega_{gc}$.
\textbf{Routh--Hurwitz.} For $a_n s^n+\cdots+a_0$: sign changes in first column of Routh array $=$ \# RHP roots. Necessary: all $a_i>0$, none missing. 2nd-order $s^2+bs+c$ stable iff $b,c>0$.
\textbf{First-order plants.} GM $=\infty$ (phase never hits $-180^\circ$). Negative feedback does \emph{not} always stabilize --- excess gain destabilizes.

\section*{CT $\leftrightarrow$ DT + Traps}

\begin{wraptable}[11]{l}{0pt}
  \begin{tabular}{@{}l l l@{}}
    \toprule
     & CT & DT \\
    \midrule
    Var & $s$ & $z$ \\
    FT on & $j\omega$ & $|z|{=}1$ \\
    ROC & strip & annulus \\
    Stable & LHP & $|p|{<}1$ \\
    Basic pair & $\tfrac{1}{s+a}$ & $\tfrac{1}{1-az^{-1}}$ \\
    Delay & $e^{-st_0}$ & $z^{-n_0}$ \\
    Unil IC & $sY{-}y(0^-)$ & $z^{-1}Y{+}y[-1]$ \\
    IVT & $\lim sX$ & $\lim X$ \\
    FVT & $\lim sX$ & $\lim(1{-}z^{-1})X$ \\
    Feedback & \multicolumn{2}{l}{$H/(1{+}GH)$} \\
    \bottomrule
  \end{tabular}
\end{wraptable}
\textbf{Traps.}
\textbf{Always state ROC.} Same $X$, different ROC $=$ different signal.
Unil.\ L: $\mathbf{-}y(0^-)$. Unil.\ Z: $\mathbf{+}y[-1]$. Opposite signs.
Repeated pole order $n$ needs $n$ PFE terms.
DT: $|a|<1$, not $a<1$. Pole at $z=-0.9$ IS stable.
FVT invalid if unstable or marginal (poles on $j\omega$ / unit circle).
$\omega_s>2\omega_M$ strict; aliasing irreversible; AAF \emph{before} the sampler.
GM read from $|L|$ at $-180^\circ$; PM from $\angle L$ at 0 dB. Don't swap.
Poles of $\tfrac{1}{1-az^{-1}}$ at $z=a$, not $1/a$.
PFE in Z: use $z^{-1}$, never $z$.
RHP \emph{zero} $\ne$ instability (only poles).
Neg fdbk $\to 1+GH$; pos $\to 1-GH$.
Negative feedback does \emph{not} always stabilize.
