\section*{The Convolution Property (Most Important)}

Convolution in time = multiplication in frequency:

\[y(t) = x(t) * h(t) \quad \longleftrightarrow \quad Y(j\omega) = X(j\omega) \cdot H(j\omega)\]

This is the \textbf{key payoff} of the frequency domain: replace difficult convolution integrals with simple multiplication.

\subsection*{Analysis Pipeline}

\begin{enumerate}
  \item Transform input: $x(t) \to X(j\omega)$
  \item Multiply: $Y(j\omega) = X(j\omega) \cdot H(j\omega)$
  \item Inverse transform: $Y(j\omega) \to y(t)$
  \item (Often use partial fractions for step 3)
\end{enumerate}

\section*{The Multiplication Property}

Multiplication in time = convolution in frequency (scaled):

\[x(t) \cdot y(t) \quad \longleftrightarrow \quad \frac{1}{2\pi} [X(j\omega) * Y(j\omega)]\]

\begin{center}
\begin{tabular}{|l|l|l|}

\hline
Differentiation (time) & $\frac{d}{dt}x(t)$ & $j\omega \, X(j\omega)$ \\

\section*{Duality}

If $x(t) \longleftrightarrow X(j\omega)$, then $X(jt) \longleftrightarrow 2\pi x(-\omega)$.

Swap time and frequency roles to get new transform pairs for free.

\section*{Time Scaling / Time-Frequency Trade-off}

\[x(at) \longleftrightarrow \frac{1}{|a|} X(j\omega/a)\]

\begin{itemize}
  \item \textbf{Compress in time} ($|a| > 1$): spectrum \textbf{expands} in frequency
  \item \textbf{Stretch in time} ($|a| < 1$): spectrum \textbf{compresses} in frequency
  \item You cannot be simultaneously narrow in both time and frequency
\end{itemize}

\section*{Conjugate Symmetry (Real Signals)}

For real $x(t)$: $X(-j\omega) = X^*(j\omega)$

This means:
\begin{itemize}
  \item $|X(j\omega)|$ is \textbf{even} (symmetric)
  \item $\angle X(j\omega)$ is \textbf{odd} (antisymmetric)
  \item $\text{Re}\{X(j\omega)\}$ is even
  \item $\text{Im}\{X(j\omega)\}$ is odd
\end{itemize}

\section*{Systems from Differential Equations}

Replace $d/dt$ with $j\omega$ to convert differential equations to algebraic equations:

\[\sum_k b_k \frac{d^k y}{dt^k} = \sum_k c_k \frac{d^k x}{dt^k} \quad \longrightarrow \quad H(j\omega) = \frac{Y(j\omega)}{X(j\omega)} = \frac{\sum_k c_k (j\omega)^k}{\sum_k b_k (j\omega)^k}\]

\section*{DT Convolution Property}

\[y[n] = x[n] * h[n] \quad \longleftrightarrow \quad Y(e^{j\omega}) = X(e^{j\omega}) \cdot H(e^{j\omega})\]

\subsection*{Systems from Difference Equations}

Replace $x[n-k]$ with $e^{-j\omega k} X(e^{j\omega})$:

\[H(e^{j\omega}) = \frac{Y(e^{j\omega})}{X(e^{j\omega})} = \frac{\sum_k c_k e^{-j\omega k}}{\sum_k b_k e^{-j\omega k}}\]
